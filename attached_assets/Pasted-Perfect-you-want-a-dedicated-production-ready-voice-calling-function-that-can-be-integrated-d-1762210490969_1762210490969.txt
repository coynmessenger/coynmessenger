# 🎙️ `initializeVoiceCall()` — Full Voice Calling Function

### ✅ Purpose

Handles:

* Voice-only media capture
* SDP negotiation (offer/answer)
* ICE candidate exchange
* Incoming/outgoing call signaling
* Automatic cleanup and reconnection safety
* Mute/unmute toggles

---

## 🔧 Implementation

```ts
// utils/voiceCall.ts
import { Socket } from "socket.io-client";

interface VoiceCallOptions {
  socket: Socket;
  localUserId: string;
  remoteUserId: string;
  callId: string;
  isCaller: boolean;
  remoteOffer?: RTCSessionDescriptionInit; // For callee
  onConnected?: () => void;
  onEnded?: () => void;
  onError?: (error: any) => void;
}

/**
 * initializeVoiceCall
 * ---------------------
 * Sets up a complete voice-only WebRTC session between two users.
 * Uses Opus codec, SRTP encryption, and Socket.io signaling.
 * 
 * @param options - configuration and callbacks for the call
 */
export async function initializeVoiceCall(options: VoiceCallOptions) {
  const {
    socket,
    localUserId,
    remoteUserId,
    callId,
    isCaller,
    remoteOffer,
    onConnected,
    onEnded,
    onError,
  } = options;

  const pc = new RTCPeerConnection({
    iceServers: [
      { urls: "stun:stun1.l.google.com:19302" },
      // Optionally add TURN for NAT/firewall traversal:
      // { urls: "turn:turn.coynful.com:3478", username: "user", credential: "pass" },
    ],
  });

  let localStream: MediaStream | null = null;
  let remoteAudio: HTMLAudioElement | null = null;

  console.log("[voiceCall] Initializing voice call", { isCaller, callId });

  try {
    // 1️⃣ Capture local microphone audio
    localStream = await navigator.mediaDevices.getUserMedia({
      audio: {
        echoCancellation: true,
        noiseSuppression: true,
        autoGainControl: true,
      },
      video: false,
    });

    // Add audio track to peer connection
    localStream.getTracks().forEach((track) => pc.addTrack(track, localStream!));

    // 2️⃣ Handle incoming remote audio track
    pc.ontrack = (event) => {
      console.log("[voiceCall] Remote audio track received");
      const [stream] = event.streams;
      remoteAudio = document.createElement("audio");
      remoteAudio.srcObject = stream;
      remoteAudio.autoplay = true;
      remoteAudio.play().catch(() => {
        console.warn("[voiceCall] Autoplay blocked, waiting for user gesture");
      });
    };

    // 3️⃣ Monitor connection state
    pc.oniceconnectionstatechange = () => {
      console.log("[voiceCall] ICE State:", pc.iceConnectionState);
      if (["connected", "completed"].includes(pc.iceConnectionState)) {
        onConnected?.();
      } else if (["failed", "disconnected"].includes(pc.iceConnectionState)) {
        console.warn("[voiceCall] Connection lost");
      }
    };

    // 4️⃣ Handle ICE candidates
    pc.onicecandidate = (event) => {
      if (event.candidate) {
        socket.emit("ice-candidate", {
          to: remoteUserId,
          from: localUserId,
          callId,
          candidate: event.candidate,
        });
      }
    };

    // 5️⃣ Negotiation logic
    if (isCaller) {
      const offer = await pc.createOffer({ offerToReceiveAudio: true });
      await pc.setLocalDescription(offer);

      socket.emit("initiate-call", {
        to: remoteUserId,
        from: localUserId,
        callId,
        sdp: pc.localDescription,
      });

      console.log("[voiceCall] Offer created and sent");
    } else {
      if (!remoteOffer) throw new Error("Missing remote offer for callee");
      await pc.setRemoteDescription(new RTCSessionDescription(remoteOffer));
      console.log("[voiceCall] Remote offer set on callee");

      const answer = await pc.createAnswer();
      await pc.setLocalDescription(answer);

      socket.emit("accept-call", {
        to: remoteUserId,
        from: localUserId,
        callId,
        sdp: pc.localDescription,
      });

      console.log("[voiceCall] Answer created and sent");
    }

    // 6️⃣ Socket event handling
    socket.on("call-accepted", async (payload) => {
      if (payload.callId !== callId) return;
      console.log("[voiceCall] Call accepted", payload);
      if (isCaller && payload.sdp) {
        await pc.setRemoteDescription(new RTCSessionDescription(payload.sdp));
      }
    });

    socket.on("ice-candidate", async (payload) => {
      if (payload.callId !== callId || !payload.candidate) return;
      try {
        await pc.addIceCandidate(new RTCIceCandidate(payload.candidate));
      } catch (err) {
        console.warn("[voiceCall] Failed to add ICE candidate", err);
      }
    });

    socket.on("call-ended", (payload) => {
      if (payload.callId !== callId) return;
      console.log("[voiceCall] Call ended remotely");
      cleanup();
      onEnded?.();
    });

  } catch (error) {
    console.error("[voiceCall] Initialization failed", error);
    onError?.(error);
    cleanup();
  }

  // 7️⃣ Helper functions
  function muteToggle() {
    if (!localStream) return;
    localStream.getAudioTracks().forEach((t) => (t.enabled = !t.enabled));
    console.log("[voiceCall] Mute toggled");
  }

  function endCall() {
    console.log("[voiceCall] Hanging up");
    socket.emit("end-call", { to: remoteUserId, callId });
    cleanup();
    onEnded?.();
  }

  function cleanup() {
    try {
      socket.off("call-accepted");
      socket.off("ice-candidate");
      socket.off("call-ended");
      if (localStream) {
        localStream.getTracks().forEach((t) => t.stop());
        localStream = null;
      }
      if (remoteAudio) {
        remoteAudio.pause();
        remoteAudio.srcObject = null;
        remoteAudio.remove();
        remoteAudio = null;
      }
      pc.close();
    } catch (e) {
      console.warn("[voiceCall] Cleanup error", e);
    }
  }

  return {
    pc,
    muteToggle,
    endCall,
    cleanup,
  };
}
```

## 🧱 Example Usage

You can call this from your `Messenger.tsx` or your call initiation flow.

```ts
import { initializeVoiceCall } from "@/utils/voiceCall";

async function startVoiceCall(socket, currentUserId, targetUserId) {
  const callId = Date.now().toString();

  const call = await initializeVoiceCall({
    socket,
    localUserId: currentUserId,
    remoteUserId: targetUserId,
    callId,
    isCaller: true,
    onConnected: () => console.log("✅ Voice connected"),
    onEnded: () => console.log("🛑 Call ended"),
    onError: (err) => console.error("❌ Voice call error", err),
  });

  // Optional: hook buttons
  document.getElementById("mute-btn")?.addEventListener("click", call.muteToggle);
  document.getElementById("hangup-btn")?.addEventListener("click", call.endCall);
}
```
