## 🧩 `server/webrtc-signaling.ts`

This is a **diagnostic version** of your signaling server that logs all call-related events.

If you already have this file, replace your event handlers with these extended logs (don’t remove your encryption logic — just wrap it with debug logs):

```ts
import express from 'express';
import http from 'http';
import { Server } from 'socket.io';

const app = express();
const server = http.createServer(app);
const io = new Server(server, { cors: { origin: '*' } });

interface ActiveCall {
  callerId: string;
  calleeId: string;
  callId: string;
}

const activeCalls: Record<string, ActiveCall> = {};

io.on('connection', (socket) => {
  console.log(`[CONNECT] Socket ${socket.id} connected`);

  socket.on('authenticate', (data) => {
    socket.data.userId = data.userId;
    socket.join(data.userId);
    console.log(`[AUTH] ${socket.id} registered as ${data.userId}`);
  });

  socket.on('initiate-call', (payload) => {
    console.log(`[CALL] ${socket.data.userId} initiating call to ${payload.to}`);
    const target = payload.to;
    const callId = payload.callId;

    activeCalls[callId] = {
      callerId: socket.data.userId,
      calleeId: target,
      callId
    };

    // Forward to callee
    io.to(target).emit('incoming-call', {
      from: socket.data.userId,
      callId,
      sdp: payload.sdp
    });

    console.log(`[FORWARD] Emitted incoming-call to ${target}`);
  });

  socket.on('accept-call', (payload) => {
    console.log(`[ACCEPT] ${socket.data.userId} accepted call from ${payload.to}`);
    io.to(payload.to).emit('call-accepted', {
      from: socket.data.userId,
      callId: payload.callId,
      sdp: payload.sdp
    });
  });

  socket.on('ice-candidate', (payload) => {
    console.log(`[ICE] ${socket.data.userId} → ${payload.to}`);
    io.to(payload.to).emit('ice-candidate', payload);
  });

  socket.on('end-call', (payload) => {
    console.log(`[END] Call ${payload.callId} ended by ${socket.data.userId}`);
    io.to(payload.to).emit('call-ended', payload);
    delete activeCalls[payload.callId];
  });

  socket.on('disconnect', () => {
    console.log(`[DISCONNECT] Socket ${socket.id} (${socket.data.userId}) disconnected`);
  });
});

const PORT = process.env.PORT || 3000;
server.listen(PORT, () => {
  console.log(`🛰️ Signaling server running on port ${PORT}`);
});
```

## 🧪 `test/simulate-calls.ts`

This simulates two clients (User A and User B) connecting, authenticating, and exchanging call events.

```ts
import { io } from 'socket.io-client';

const URL = 'http://localhost:3000';
const caller = io(URL);
const callee = io(URL);

caller.on('connect', () => {
  console.log('[CALLER] Connected', caller.id);
  caller.emit('authenticate', { userId: 'userA' });

  // Wait to ensure callee is ready
  setTimeout(() => {
    console.log('[CALLER] Initiating call to userB...');
    caller.emit('initiate-call', {
      to: 'userB',
      sdp: 'fake-offer',
      callId: Date.now().toString()
    });
  }, 1500);
});

callee.on('connect', () => {
  console.log('[CALLEE] Connected', callee.id);
  callee.emit('authenticate', { userId: 'userB' });

  callee.on('incoming-call', (payload) => {
    console.log('[CALLEE] Incoming call detected:', payload);
    // Simulate acceptance
    callee.emit('accept-call', {
      to: 'userA',
      sdp: 'fake-answer',
      callId: payload.callId
    });
  });
});

caller.on('call-accepted', (payload) => {
  console.log('[CALLER] Call accepted:', payload);
});

caller.on('ice-candidate', (data) => console.log('[CALLER] ICE from callee', data));
callee.on('ice-candidate', (data) => console.log('[CALLEE] ICE from caller', data));

caller.on('call-ended', (payload) => console.log('[CALLER] Call ended:', payload));
callee.on('call-ended', (payload) => console.log('[CALLEE] Call ended:', payload));

caller.onAny((event, ...args) => console.log('[CALLER EVENT]', event, args));
callee.onAny((event, ...args) => console.log('[CALLEE EVENT]', event, args));
```
